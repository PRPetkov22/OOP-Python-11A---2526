class Student:
    def __init__(self, name, grades):
        self.name = name
        self.grades = grades

    def average(self):
        return sum(self.grades) / len(self.grades)


student = Student("Пламен", [2, 4, 3, 6, 6, 5, 4])

print(f"Име - {student.name}")

print(f"Среден успех - {student.average():.2f}")
