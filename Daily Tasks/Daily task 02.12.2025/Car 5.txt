class Car:
    def __init__(self, model, usage, fuel):
        self.model = model
        self.usage = usage
        self.fuel = fuel

    def can_travel(self, km):
        need_fuel = (km / 100) * self.usage
        if need_fuel <= self.fuel:
            return True
        else:
            return False


Car2 = Car("Nissan", 15, 60)

print(f"Моделът на колата е {Car2.model}")

print(f"Разходът на колата е {Car2.usage} литра на 100км")

print(f"Горивото в резервоара е {Car2.fuel} литра")

print()


kilom = 1300
if Car2.can_travel(kilom):
    print(f"Колата може да измине {kilom} км")
else:
    print(f"Колата не може да измине {kilom} км")
