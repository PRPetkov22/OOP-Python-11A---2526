class PasswordChecker:
    def __init__(self, password):
        self.password = password

    def strength(self):
        p = self.password

        lower = any(c.islower() for c in p)
        upper = any(c.isupper() for c in p)
        digit = any(c.isdigit() for c in p)
        special_char = any(not c.isalnum() for c in p)

        if len(p) < 8:
            return "Слаба"

        if len(p) >= 8 and digit and not (lower and upper and special_char):
            return "Средна"

        if lower and upper and digit and special_char:
            return "Силна"

        return "Средна"


password_1 = PasswordChecker("Mancho")

password_2 = PasswordChecker("Mancho123")

password_3 = PasswordChecker("Mancho#mnqk777")

print(password_1.strength())

print(password_2.strength())

print(password_3.strength())
