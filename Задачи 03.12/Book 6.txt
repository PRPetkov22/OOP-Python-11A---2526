class Book:
    def __init__(self, title, author, pages):
        self.title = title
        self.author = author
        self.pages = pages

    def info(self):
        return f"Заглавие - {self.title}, Автор - {self.author}, Страници - {self.pages}"

book1 = Book("Под игото", "Иван Вазов", 520)

book2 = Book("TO", "Стивън Кинг", 280)

print(book1.info())

print(book2.info())
